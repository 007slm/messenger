Messenger - 跨域 Iframe 通信演示

演示

我是父页面 

来自子页面的消息:

父页面这么写

seajs.use(['messenger', '$'], function(Messenger, $) {
    var messenger = new Messenger({
        target: '#iframe',
        onmessage: function(data) {
            output.html(output.html() + ' ' + data);
        }
    });

    // 通过点击按钮发送消息
    var send = $('#send');
    var message = $('#message');
    var output = $('#output');
    send.click(function() {
        messenger.send(message[0].value);
        message[0].value = '';
    });
});
